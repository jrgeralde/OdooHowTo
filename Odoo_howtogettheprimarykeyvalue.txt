from odoo import models, api

class MyClass(models.Model):
    _name = 'my.module'
    
    @api.model
    def get_table_id(self):
        table_id = self.env['ir.model'].search([('model', '=', 'my.module')], limit=1)
        if table_id:
            return table_id.id
        else:
            return False
            
 # to get the primary key id value

if table_id:

else:
    print("Table not found.")
